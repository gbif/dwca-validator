18  /**
19   * ChainableRecordEvaluator to check the values inside a record.
20   * This validation is about what the value is and not what the value represents.
21   * TODO: Evaluate the possibility to add its own builder and offer more options like applyToAll
22   * 
23   * @author cgendreau
24   */
25  public class ValueEvaluator implements RecordEvaluatorIF {
26  
27    // hold all evaluation rules per ConceptTerm
28    private final Map<ConceptTerm, List<EvaluationRuleIF<String>>> rulesPerTerm;
29    private final ValidationContext evaluatorContext;
30  
31    public ValueEvaluator(Map<ConceptTerm, List<EvaluationRuleIF<String>>> rulesPerTerm,
32      ValidationContext evaluatorContext) {
33      this.rulesPerTerm =
34        Collections.unmodifiableMap(new HashMap<ConceptTerm, List<EvaluationRuleIF<String>>>(rulesPerTerm));
35      this.evaluatorContext = evaluatorContext;
36    }
37  
38    @Override
39    public void handleEval(Record record, ResultAccumulatorIF resultAccumulator) {
40      List<ValidationResultElement> elementList = null;
41      ValidationResultElement validationResultElement;
42  
43      // only iterate over terms we have a rule for
44      for (ConceptTerm currTerm : rulesPerTerm.keySet()) {
45        for (EvaluationRuleIF<String> currRule : rulesPerTerm.get(currTerm)) {
46          validationResultElement = currRule.evaluate(record.value(currTerm));
47          if (validationResultElement != null) {
48            // lazy create the list assuming, in normal case, we should have more valid record
49            if (elementList == null) {
50              elementList = new ArrayList<ValidationResultElement>();
51            }
52            elementList.add(validationResultElement);
53          }
54        }
55      }
56  
57      if (elementList != null && elementList.size() > 0) {
58        resultAccumulator.accumulate(new ValidationResult(record.id(), evaluatorContext, elementList));
59      }
60    }
61  
62    @Override
63    public void postIterate() {
64      // noop
65    }
66  }
