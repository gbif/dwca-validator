001package org.gbif.dwc.validator.result.impl;
002
003import org.gbif.dwc.validator.result.ResultAccumulatorIF;
004import org.gbif.dwc.validator.result.ValidationResult;
005
006import java.io.File;
007import java.io.FileWriter;
008import java.io.IOException;
009import java.util.Iterator;
010import java.util.concurrent.ConcurrentLinkedQueue;
011import java.util.concurrent.atomic.AtomicBoolean;
012import java.util.concurrent.atomic.AtomicInteger;
013
014import org.slf4j.Logger;
015import org.slf4j.LoggerFactory;
016
017/**
018 * ResultAccumulatorIF implementation that use a threshold to determine when to flush to a file.
019 * 
020 * @author cgendreau
021 */
022public class ThresholdResultAccumulator implements ResultAccumulatorIF {
023
024  private static final Logger LOGGER = LoggerFactory.getLogger(ThresholdResultAccumulator.class);
025  protected static final int DEFAULT_THRESHOLD = 1000;
026  private final ConcurrentLinkedQueue<ValidationResult> queue;
027
028  private final FileWriter fw;
029  private final int threshold;
030  private AtomicInteger count = new AtomicInteger();
031  private final AtomicBoolean flushing = new AtomicBoolean();
032
033  public ThresholdResultAccumulator(String filePath) throws IOException {
034    this(filePath, DEFAULT_THRESHOLD);
035  }
036
037  public ThresholdResultAccumulator(String filePath, int threshold) throws IOException {
038    fw = new FileWriter(new File(filePath));
039    this.threshold = threshold;
040    queue = new ConcurrentLinkedQueue<ValidationResult>();
041    count = new AtomicInteger(0);
042  }
043
044  /**
045   * This method is non-blocking except when a flush needs to be perform.
046   * 
047   * @param result
048   * @return
049   */
050  @Override
051  public boolean accumulate(ValidationResult result) {
052    queue.add(result);
053
054    if (count.incrementAndGet() == threshold && flushing.compareAndSet(false, true)) {
055      flush(threshold);
056    }
057    return true;
058  }
059
060  @Override
061  public void close() {
062    try {
063
064      flushAll();
065
066      fw.flush();
067      fw.close();
068    } catch (IOException ioEx) {
069      LOGGER.error("Can't close NonBlockingResultAccumulator properly", ioEx);
070    }
071
072  }
073
074  /**
075   * Flush a specific number of records to the file.
076   * 
077   * @param howMany
078   */
079  private void flush(int howMany) {
080    int numberWritten = 0;
081    ValidationResult currentResult = queue.poll();
082    String resultLine = null;
083    try {
084      while (currentResult != null && (numberWritten < howMany)) {
085        resultLine = currentResult.getId();
086        fw.write(resultLine);
087        numberWritten++;
088
089        currentResult = queue.poll();
090      }
091    } catch (IOException ioEx) {
092      LOGGER.error("Can't flush to file using FileWriter", ioEx);
093    }
094    flushing.set(false);
095  }
096
097  /**
098   * Flush all remaining content of the queue to the file.
099   */
100  private void flushAll() {
101    Iterator<ValidationResult> queueIterator = queue.iterator();
102    ValidationResult currentResult = null;
103    try {
104      while (queueIterator.hasNext()) {
105        currentResult = queueIterator.next();
106        queueIterator.remove();
107        String resultLine = currentResult.getId();
108        fw.write(resultLine);
109      }
110    } catch (IOException ioEx) {
111      LOGGER.error("Can't flush to file using FileWriter", ioEx);
112    }
113  }
114
115  @Override
116  public int getCount() {
117    return count.get();
118  }
119
120}
