001package org.gbif.dwc.validator.evaluator;
002
003import org.gbif.dwc.validator.evaluator.impl.UniquenessEvaluator;
004import org.gbif.dwc.validator.evaluator.impl.ValueEvaluator;
005
006/**
007 * Build evaluator chains.
008 * TODO allow concrete ChainableRecordEvaluator customizations
009 * TODO allow DI to set concrete ChainableRecordEvaluator
010 * 
011 * @author cgendreau
012 */
013public class EvaluatorChainBuilder {
014
015
016  /**
017   * Build a validation chain for DarwinCore core component.
018   * 
019   * @return
020   */
021  public static ChainableRecordEvaluator buildCoreChain() {
022    return linkEvaluators(new ValueEvaluator(), new UniquenessEvaluator());
023  }
024
025  /**
026   * Links ChainableRecordEvaluator together.
027   * 
028   * @param evaluators
029   * @return head of the chain
030   */
031  private static ChainableRecordEvaluator linkEvaluators(ChainableRecordEvaluator... evaluators) {
032    ChainableRecordEvaluator previousEvaluator = null;
033    for (ChainableRecordEvaluator currEvaluator : evaluators) {
034      if (previousEvaluator != null) {
035        previousEvaluator.setNextElement(currEvaluator);
036      }
037      previousEvaluator = currEvaluator;
038    }
039    return evaluators[0];
040  }
041}
