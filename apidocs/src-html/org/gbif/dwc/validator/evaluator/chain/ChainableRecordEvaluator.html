001package org.gbif.dwc.validator.evaluator.chain;
002
003import org.gbif.dwc.record.Record;
004import org.gbif.dwc.validator.evaluator.RecordEvaluatorIF;
005import org.gbif.dwc.validator.result.ResultAccumulatorIF;
006
007/**
008 * Wrapper class to allow RecordEvaluatorIF to be chainable.
009 * The class is immutable but the RecordEvaluatorIF immutability can not be enforced.
010 * 
011 * @author cgendreau
012 */
013public class ChainableRecordEvaluator {
014
015  private final RecordEvaluatorIF recordEvaluator;
016  private final ChainableRecordEvaluator nextElement;
017
018  public ChainableRecordEvaluator(RecordEvaluatorIF recordEvaluator, ChainableRecordEvaluator nextElement) {
019    this.recordEvaluator = recordEvaluator;
020    this.nextElement = nextElement;
021  }
022
023  /**
024   * Do validation and call next element in the chain (if there is one).
025   * 
026   * @param record
027   * @param resultAccumulator
028   */
029  public void doEval(Record record, ResultAccumulatorIF resultAccumulator) {
030    recordEvaluator.handleEval(record, resultAccumulator);
031    if (nextElement != null) {
032      nextElement.doEval(record, resultAccumulator);
033    }
034  }
035}
