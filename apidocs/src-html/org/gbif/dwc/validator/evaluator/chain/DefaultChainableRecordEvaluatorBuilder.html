001package org.gbif.dwc.validator.evaluator.chain;
002
003import org.gbif.dwc.validator.evaluator.RecordEvaluatorIF;
004
005import java.util.ArrayList;
006import java.util.List;
007
008/**
009 * Default ChainableRecordEvaluatorBuilderIF implementation.
010 * 
011 * @author cgendreau
012 */
013public class DefaultChainableRecordEvaluatorBuilder implements ChainableRecordEvaluatorBuilderIF {
014
015  private final List<RecordEvaluatorIF> recordEvaluatorList;
016
017  private DefaultChainableRecordEvaluatorBuilder(RecordEvaluatorIF recordEvaluator,
018    List<RecordEvaluatorIF> recordEvaluatorList) {
019    this.recordEvaluatorList = new ArrayList<RecordEvaluatorIF>(recordEvaluatorList);
020    this.recordEvaluatorList.add(recordEvaluator);
021  }
022
023  /**
024   * Create a builder from the provided head RecordEvaluatorIF
025   * 
026   * @param head
027   * @return
028   */
029  public static DefaultChainableRecordEvaluatorBuilder create(RecordEvaluatorIF head) {
030    return new DefaultChainableRecordEvaluatorBuilder(head, new ArrayList<RecordEvaluatorIF>());
031  }
032
033  @Override
034  public ChainableRecordEvaluator build() {
035    ChainableRecordEvaluator current = null;
036    // iterate from last element to the first to be able to provide the next ChainableRecordEvaluator
037    for (int i = recordEvaluatorList.size() - 1; i >= 0; i--) {
038      current = new ChainableRecordEvaluator(recordEvaluatorList.get(i), current);
039    }
040    return current;
041  }
042
043  @Override
044  public ChainableRecordEvaluatorBuilderIF linkTo(RecordEvaluatorIF recordEvaluator) {
045    return new DefaultChainableRecordEvaluatorBuilder(recordEvaluator, this.recordEvaluatorList);
046  }
047}
