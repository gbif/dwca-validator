001package org.gbif.dwc.validator.evaluator.chain;
002
003import org.gbif.dwc.terms.ConceptTerm;
004import org.gbif.dwc.terms.DwcTerm;
005import org.gbif.dwc.validator.evaluator.impl.UniquenessEvaluator;
006import org.gbif.dwc.validator.evaluator.impl.ValueEvaluator;
007import org.gbif.dwc.validator.result.ValidationContext;
008import org.gbif.dwc.validator.rule.EvaluationRuleIF;
009import org.gbif.dwc.validator.rule.value.InvalidCharacterEvaluationRule;
010
011import java.util.ArrayList;
012import java.util.HashMap;
013import java.util.List;
014import java.util.Map;
015
016/**
017 * Default implementation of EvaluationChainProviderIF using the default builder.
018 * 
019 * @author cgendreau
020 */
021public class DefaultEvaluationChainProvider implements EvaluationChainProviderIF {
022
023  /**
024   * Build a default Map<ConceptTerm, List<EvaluationRuleIF<String>>> for ValueEvaluator.
025   * TODO Evaluate the possibility to move the creation of Evaluators in their own class.
026   * 
027   * @return
028   */
029  private Map<ConceptTerm, List<EvaluationRuleIF<String>>> buildDefaultValueEvaluatorRulesPerTermMap() {
030    Map<ConceptTerm, List<EvaluationRuleIF<String>>> rulesPerTerm =
031      new HashMap<ConceptTerm, List<EvaluationRuleIF<String>>>();
032
033    // register an InvalidCharacterEvaluationRule for scientificName
034    List<EvaluationRuleIF<String>> ruleList = new ArrayList<EvaluationRuleIF<String>>();
035    ruleList.add(InvalidCharacterEvaluationRule.createRule().build());
036    rulesPerTerm.put(DwcTerm.scientificName, ruleList);
037
038    return rulesPerTerm;
039  }
040
041  @Override
042  public ChainableRecordEvaluator getCoreChain() {
043    return DefaultChainableRecordEvaluatorBuilder
044      .create(new ValueEvaluator(buildDefaultValueEvaluatorRulesPerTermMap(), ValidationContext.CORE))
045      .linkTo(new UniquenessEvaluator()).build();
046  }
047
048}
