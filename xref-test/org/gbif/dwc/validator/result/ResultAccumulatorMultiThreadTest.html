46    /**
47     * Create and run testing thread(s)
48     * 
49     * @param threadCount
50     * @throws InterruptedException
51     * @throws ExecutionException
52     */
53    private void testThread(final ResultAccumulatorIF resultAccumulator, final int threadCount)
54      throws InterruptedException, ExecutionException {
55  
56      List<Callable<Boolean>> tasks = new ArrayList<Callable<Boolean>>();
57      for (int i = 0; i < threadCount; i++) {
58        // new set of random data for each thread but same FileWriterResultAccumulator.
59        final List<String> dummyIdList = MockDataGenerator.newRandomDataList(NUMBER_OF_DATA, 4);
60  
61        Callable<Boolean> task = new Callable<Boolean>() {
62  
63          @Override
64          public Boolean call() {
65            boolean success = true;
66            for (String currDummyId : dummyIdList) {
67              success = (success && resultAccumulator.accumulate(new ValidationResult(currDummyId, null, null)));
68            }
69            return success;
70          }
71        };
72        tasks.add(task);
73      }
74  
75      ExecutorService executorService = Executors.newFixedThreadPool(threadCount);
76      // call all threads and wait for completion
77      List<Future<Boolean>> futures = executorService.invokeAll(tasks);
78  
79      resultAccumulator.close();
80      // Validate
81      Assert.assertEquals(futures.size(), threadCount);
82      Assert.assertEquals(threadCount * NUMBER_OF_DATA, resultAccumulator.getCount());
83    }
