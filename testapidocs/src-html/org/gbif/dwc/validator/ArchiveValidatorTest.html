001package org.gbif.dwc.validator;
002
003import org.gbif.dwc.text.Archive;
004import org.gbif.dwc.validator.handler.ArchiveContentHandler;
005import org.gbif.dwc.validator.handler.ArchiveStructureHandler;
006import org.gbif.dwc.validator.impl.ArchiveValidator;
007import org.gbif.dwc.validator.result.ResultAccumulatorIF;
008import org.gbif.metadata.eml.Eml;
009
010import java.io.File;
011import java.io.IOException;
012import java.net.URISyntaxException;
013
014import org.apache.commons.io.FileUtils;
015import org.junit.Test;
016
017import static org.junit.Assert.assertTrue;
018import static org.junit.Assert.fail;
019
020/**
021 * Unit tests related to ArchiveValidator.
022 * 
023 * @author cgendreau
024 */
025public class ArchiveValidatorTest {
026
027  private static class MockArchiveStructureHandler extends ArchiveStructureHandler {
028
029    private boolean inspectArchiveContentCalled = false;
030    private boolean inspectEMLCalled = false;
031    private boolean inspectMetaXMLCalled = false;
032
033    public boolean allInspectMethodsCalled() {
034      return (inspectArchiveContentCalled && inspectEMLCalled && inspectMetaXMLCalled);
035    }
036
037    @Override
038    public void inspectArchiveContent(Archive archive, ResultAccumulatorIF resultAccumulator) {
039      super.inspectArchiveContent(archive, resultAccumulator);
040      inspectArchiveContentCalled = true;
041    }
042
043    @Override
044    public void inspectEML(Eml eml, ResultAccumulatorIF resultAccumulator) {
045      super.inspectEML(eml, resultAccumulator);
046      inspectEMLCalled = true;
047    }
048
049    @Override
050    public void inspectMetaXML(File metaXML, ResultAccumulatorIF resultAccumulator) {
051      super.inspectMetaXML(metaXML, resultAccumulator);
052      inspectMetaXMLCalled = true;
053    }
054  }
055
056  /**
057   * Simply test that we can extract anm archive and call the appropriate method in ArchiveValidator.
058   */
059  @Test
060  public void testValidateArchive() {
061    ArchiveValidator validator = new ArchiveValidator();
062    MockArchiveStructureHandler structureHandler = new MockArchiveStructureHandler();
063
064    File testDwcFolder = new File("test-dwca-" + System.currentTimeMillis());
065    testDwcFolder.mkdir();
066    validator.setWorkingFolder(testDwcFolder.getAbsolutePath());
067    validator.setStructureHandler(structureHandler);
068    validator.setContentHandler(new ArchiveContentHandler());
069
070    try {
071      File testDwca = new File(getClass().getResource("/dwca/vascan_dwca.zip").toURI());
072      validator.validateArchive(testDwca);
073      // this archive does not contain EML file
074      assertTrue(structureHandler.inspectArchiveContentCalled && structureHandler.inspectMetaXMLCalled);
075    } catch (URISyntaxException e) {
076      e.printStackTrace();
077      fail();
078    }
079
080    // delete test folder
081    try {
082      FileUtils.deleteDirectory(testDwcFolder);
083    } catch (IOException e) {
084      e.printStackTrace();
085    }
086  }
087
088}
