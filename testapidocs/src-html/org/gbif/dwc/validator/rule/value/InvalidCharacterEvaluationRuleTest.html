001package org.gbif.dwc.validator.rule.value;
002
003import org.junit.Test;
004
005import static org.junit.Assert.assertNotNull;
006import static org.junit.Assert.assertNull;
007
008/**
009 * Ensure InvalidCharacterEvaluationRule object obtained by the builder work as expected.
010 * 
011 * @author cgendreau
012 */
013public class InvalidCharacterEvaluationRuleTest {
014
015  // never allowed char
016  private static final char NULL_CHAR = 0;
017  private static final char ESCAPE_CHAR = 27;
018
019  // allowed if allowFormattingWhiteSpace() is used
020  private static final String ENDLINE = System.getProperty("line.separator");
021  private static final char VERTICAL_TAB_CHAR = 11;
022
023  @Test
024  public void evaluateFormattingWhiteSpaceAllowed() {
025    InvalidCharacterEvaluationRule rule =
026      InvalidCharacterEvaluationRule.createRule().allowFormattingWhiteSpace().build();
027
028    testAlwaysValidString(rule);
029    testNeverValidString(rule);
030
031    assertNull(rule.evaluate("test" + VERTICAL_TAB_CHAR));
032  }
033
034  @Test
035  public void evaluateNoFormattingWhiteSpaceAllowed() {
036    InvalidCharacterEvaluationRule rule = InvalidCharacterEvaluationRule.createRule().build();
037
038    testAlwaysValidString(rule);
039    testNeverValidString(rule);
040
041    assertNotNull(rule.evaluate("test\t2"));
042    assertNotNull(rule.evaluate("test" + ENDLINE));
043  }
044
045  private void testAlwaysValidString(InvalidCharacterEvaluationRule rule) {
046    assertNull(rule.evaluate("test"));
047    assertNull(rule.evaluate("test 2"));
048  }
049
050  private void testNeverValidString(InvalidCharacterEvaluationRule rule) {
051    assertNotNull(rule.evaluate("test" + NULL_CHAR));
052    assertNotNull(rule.evaluate("test" + ESCAPE_CHAR));
053  }
054}
