001package org.gbif.dwc.validator.result;
002
003import org.gbif.dwc.validator.mock.MockDataGenerator;
004import org.gbif.dwc.validator.result.impl.FileWriterResultAccumulator;
005import org.gbif.dwc.validator.result.impl.ThresholdResultAccumulator;
006
007import java.io.File;
008import java.io.IOException;
009import java.util.List;
010
011import org.junit.Assert;
012import org.junit.Test;
013
014/**
015 * Test regular usage of different ResultAccumulatorIF with a single thread.
016 * 
017 * @author cgendreau
018 */
019public class ResultAccumulatorTest {
020
021  public static final int NUMBER_OF_RECORDS = 100000;
022
023  @Test
024  public void testFileWriterResultAccumulator() {
025    long t = System.currentTimeMillis();
026    String fileName = "test_FileWriterResultAccumulator.txt";
027    ResultAccumulatorIF fwra = null;
028    try {
029      fwra = new FileWriterResultAccumulator(fileName);
030    } catch (IOException e) {
031      e.printStackTrace();
032    }
033    List<String> dummyIdList = MockDataGenerator.newRandomDataList(NUMBER_OF_RECORDS, 4);
034    testResultAccumulator(fwra, dummyIdList);
035    System.out.println("Using FileWriterResultAccumulator: " + (System.currentTimeMillis() - t) + " ms");
036
037    Assert.assertEquals(NUMBER_OF_RECORDS, fwra.getCount());
038    // clean up
039    new File(fileName).delete();
040  }
041
042  private void testResultAccumulator(ResultAccumulatorIF fwra, List<String> dummyIdList) {
043    for (String currDummyId : dummyIdList) {
044      fwra.accumulate(new ValidationResult(currDummyId, null, null));
045    }
046    fwra.close();
047  }
048
049  @Test
050  public void testThresholdResultAccumulator() {
051    long t = System.currentTimeMillis();
052    String fileName = "test_ThresholdResultAccumulator.txt";
053    ResultAccumulatorIF tra = null;
054    try {
055      tra = new ThresholdResultAccumulator(fileName);
056    } catch (IOException e) {
057      e.printStackTrace();
058    }
059    List<String> dummyIdList = MockDataGenerator.newRandomDataList(NUMBER_OF_RECORDS, 4);
060    testResultAccumulator(tra, dummyIdList);
061    System.out.println("Using ThresholdResultAccumulator: " + (System.currentTimeMillis() - t) + " ms");
062    Assert.assertEquals(NUMBER_OF_RECORDS, tra.getCount());
063    // clean up
064    new File(fileName).delete();
065  }
066
067}
