001package org.gbif.dwc.validator.result;
002
003import org.gbif.dwc.validator.mock.MockDataGenerator;
004import org.gbif.dwc.validator.result.impl.FileWriterResultAccumulator;
005import org.gbif.dwc.validator.result.impl.ThresholdResultAccumulator;
006
007import java.io.File;
008import java.io.IOException;
009import java.util.ArrayList;
010import java.util.List;
011import java.util.concurrent.Callable;
012import java.util.concurrent.ExecutionException;
013import java.util.concurrent.ExecutorService;
014import java.util.concurrent.Executors;
015import java.util.concurrent.Future;
016
017import org.junit.Assert;
018import org.junit.Test;
019
020/**
021 * Test FileWriterResultAccumulator in a multi-thread context.
022 * 
023 * @author cgendreau
024 */
025public class ResultAccumulatorMultiThreadTest {
026
027  private static final int NUMBER_OF_DATA = 10000;
028
029  @Test
030  public void testFileWriterResultAccumulator16Threads() throws InterruptedException, ExecutionException {
031    long t = System.currentTimeMillis();
032    ResultAccumulatorIF fwra = null;
033    String fileName = "test_FileWriterResultAccumulator16Threads.txt";
034    try {
035      fwra = new FileWriterResultAccumulator(fileName);
036    } catch (IOException e) {
037      e.printStackTrace();
038    }
039
040    testThread(fwra, 16);
041    System.out.println("FileWriterResultAccumulator took " + (System.currentTimeMillis() - t) + " ms");
042    // clean up
043    new File(fileName).delete();
044  }
045
046  /**
047   * Create and run testing thread(s)
048   * 
049   * @param threadCount
050   * @throws InterruptedException
051   * @throws ExecutionException
052   */
053  private void testThread(final ResultAccumulatorIF resultAccumulator, final int threadCount)
054    throws InterruptedException, ExecutionException {
055
056    List<Callable<Boolean>> tasks = new ArrayList<Callable<Boolean>>();
057    for (int i = 0; i < threadCount; i++) {
058      // new set of random data for each thread but same FileWriterResultAccumulator.
059      final List<String> dummyIdList = MockDataGenerator.newRandomDataList(NUMBER_OF_DATA, 4);
060
061      Callable<Boolean> task = new Callable<Boolean>() {
062
063        @Override
064        public Boolean call() {
065          boolean success = true;
066          for (String currDummyId : dummyIdList) {
067            success = (success && resultAccumulator.accumulate(new ValidationResult(currDummyId, null, null)));
068          }
069          return success;
070        }
071      };
072      tasks.add(task);
073    }
074
075    ExecutorService executorService = Executors.newFixedThreadPool(threadCount);
076    // call all threads and wait for completion
077    List<Future<Boolean>> futures = executorService.invokeAll(tasks);
078
079    resultAccumulator.close();
080    // Validate
081    Assert.assertEquals(futures.size(), threadCount);
082    Assert.assertEquals(threadCount * NUMBER_OF_DATA, resultAccumulator.getCount());
083  }
084
085  @Test
086  public void testThresholdResultAccumulator16Threads() throws InterruptedException, ExecutionException {
087    long t = System.currentTimeMillis();
088    ResultAccumulatorIF ra = null;
089    String fileName = "test_ThresholdResultAccumulator16Threads.txt";
090    try {
091      ra = new ThresholdResultAccumulator(fileName, 1000);
092    } catch (IOException e) {
093      e.printStackTrace();
094    }
095
096    testThread(ra, 16);
097    System.out.println("ThresholdResultAccumulator took " + (System.currentTimeMillis() - t) + " ms");
098    // clean up
099    new File(fileName).delete();
100  }
101}
