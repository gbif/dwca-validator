026/**
027 * Test ValueEvaluator implementation
028 * 
029 * @author cgendreau
030 */
031public class ValueEvaluatorTest {
032
033  private Record buildMockRecord(String id) {
034    // create a mock Record
035    ArchiveField idField = new ArchiveField(0, DwcTerm.occurrenceID, null, DataType.string);
036    ArchiveField scientificNameField = new ArchiveField(1, DwcTerm.scientificName, null, DataType.string);
037    List<ArchiveField> fieldList = new ArrayList<ArchiveField>();
038    fieldList.add(scientificNameField);
039    RecordImpl testRecord = new RecordImpl(idField, fieldList, "rowType", false);
040    testRecord.setRow(new String[] {id, "gulo\tgulo"});
041    return testRecord;
042  }
043
044  @Test
045  public void testValueEvaluator() {
046
047    InMemoryResultAccumulator resultAccumulator = new InMemoryResultAccumulator();
048
049    Map<ConceptTerm, List<EvaluationRuleIF<String>>> rulesPerTerm =
050      new HashMap<ConceptTerm, List<EvaluationRuleIF<String>>>();
051
052    // register an InvalidCharacterEvaluationRule for scientificName
053    List<EvaluationRuleIF<String>> ruleList = new ArrayList<EvaluationRuleIF<String>>();
054    ruleList.add(InvalidCharacterEvaluationRule.createRule().build());
055    rulesPerTerm.put(DwcTerm.scientificName, ruleList);
056
057    ValueEvaluator valueEvaluator = new ValueEvaluator(rulesPerTerm, ValidationContext.CORE);
058    // clear original map to ensure ValueEvaluator immutability
059    rulesPerTerm.clear();
060
061    valueEvaluator.handleEval(buildMockRecord("1"), resultAccumulator);
062
063    assertTrue(resultAccumulator.getValidationResultsList().size() > 0);
064    assertEquals("1", resultAccumulator.getValidationResultsList().get(0).getId());
065  }
066
067}
